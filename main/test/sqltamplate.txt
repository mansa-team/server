CREATE TABLE IF NOT EXISTS stocksapi_keys (
    apiKey VARCHAR(255) PRIMARY KEY,
    userId INT NOT NULL UNIQUE,
    requestLimit INT NOT NULL,
    currentUsage INT DEFAULT 0,
    lastReset TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
